DOM节点

节点操作示例

window.onload = function () {
var oUl=document.getElementById('ul1');
/*alert(oUl.childNodes.length); //IE6-8显示为5个节点，其他的浏览器显示为11个节点*/
for(var i=0;i< oUl.childNodes.length; i ++)
{
/*oUl.childNodes[i].style.background='red'; //这样设置是不起作用的。
oUl.childNodes[0].style不存在，他是一个文本节点，也是一个无法选中的。
*/
/*alert(oUl.childNodes[i].nodeType);
nodeType==3; 文本节点 nodeType()节点类型
nodeType==1； 元素节点
*/
if(oUl.childNodes[i].nodeType==1)
{
oUl.childNodes[i].style.background='red';
/* 这种写法适用于所有的浏览器*/
}
}
alert(oUl.children.length);//children这个属性直接返回元素节点，使用更方便。
}

注意点

childNodes和children都只是选中了子元素节点不会选中子元素下的其他元素
